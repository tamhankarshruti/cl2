# 16-Qubit Quantum Random Number Generator (text drawer – no extra installs)
from qiskit import QuantumCircuit, transpile
#Used to create and manipulate quantum circuits — the core of Qiskit.
#Optimizes your circuit to run efficiently on a specific backend (simulator
from qiskit_aer import AerSimulator 
#A simulator backend that mimics a quantum computer, allowing to run circuits without real quantum hardware

# Create a quantum circuit with 16 qubits and classical bits
circuit = QuantumCircuit(16, 16)    #(quantum bits, classical bits)
#Classical bits are used to store measurement results (0s and 1s).
circuit.h(range(16))      # Applies a Hadamard gate (H) to each of the 16 qubits, Put all qubits in superposition
circuit.measure(range(16), range(16))  #after measuring, you get a 16-bit random binary number

# Text-based circuit diagram (prints to the notebook/console)
print("\nQuantum Circuit:")
print(circuit.draw(output='text'))    # <-- no matplotlib, no pylatexenc

# Simulate the quantum circuit
simulator = AerSimulator()
compiled_circuit = transpile(circuit, simulator)

job = simulator.run(compiled_circuit, shots=1)
result = job.result()
counts = result.get_counts()

# Extract the random number
random_number = int(list(counts.keys())[0], 2)
binary_number = bin(random_number)[2:].zfill(16)

print("\nRandom number (decimal):", random_number)
print("Random number (binary) :", binary_number)
